\small\underline{\textbf{Hyperbolic Functions:}}


\begin{gather*}
\sinh(x) = \dfrac{e^{x} - e^{-x}}{2} \\
\cosh(x) = \dfrac{e^{x} + e^{-x}}{2} \\
\tanh(x) = \dfrac{\sinh(x)}{\cosh(x)}
\end{gather*}

\small\underline{\textbf{Hyperbolic Identities:}}

\begin{gather*}
\cosh^{2}(x) - \sinh^{2}(x) = 1 \\
\cosh(2x) - 1 = 2\sinh^{2}(x) \\
\sinh(x) + \cosh(x) = e^{x}
\end{gather*}


\small\underline{\textbf{Trigonometric Identities:}}

\begin{gather*}
\sin^{2}(x) + \cos^{2}(x) = 1 \\
\sec^{2}(x) - \tan^{2}(x) = 1 \\
\tan^{2}(x) + 1 = \sec^{2}(x) \\
\sin(2x) = 2\sin(x)\cos(x)
\end{gather*}




\small\underline{\textbf{Logarithm Properties:}}

\begin{align*}
\ln(ab) &= \ln(a) + \ln(b) \\
\ln(a^{n}) &= n\,\ln(a) \\
\log_{b}(x) &= \dfrac{\ln(x)}{\ln(b)}
\end{align*}
